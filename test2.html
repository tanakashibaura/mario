def _y_reward(self):
    if self._y_position < 79:
       return 0
    _reward = (self._y_position - self._y_position_last) / 10
    self._y_position_last = self._y_position
    return _reward
これを下のように_rewardの値を増やした. 
def _y_reward(self):
    if self._y_position < 79:
        return 0
    _reward = (self._y_position - self._y_position_last) / 5
    self._y_position_last = self._y_position
    return _reward
